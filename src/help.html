How can I add a big script or a linked script, like jQuery?

You can always use javascript to load an external Javascript URL into your page.
You can copy paste the following code for a try: 

function loadScript(url, callback)
{
// Adding the script tag to the head as suggested before
var head = document.getElementsByTagName('head')[0];
var script = document.createElement('script');
script.type = 'text/javascript';
script.src = url;

// Then bind the event to the callback function.
// There are several events for cross browser compatibility.
script.onreadystatechange = callback;
script.onload = callback;

// Fire the loading
head.appendChild(script);
}

loadScript('https://code.jquery.com/jquery-1.11.1.min.js', function() {});
